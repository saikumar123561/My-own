public class Rou {

	String name = "sai"; 
	String n = new String("saikumar");
	String n3 = new String("saikumar");
	String nam = "sai";
	String n1 = new String("sai");
	String n2 = new String("sai");
	
	public void verify() {
		System.out.println(name==nam);    //true
		System.out.println(n==nam);       //false
		System.out.println(n==n1);        //false
		System.out.println(n.equals(n1));     //false
		System.out.println(n1.equals(n2));    //true
		System.out.println(n.equals(n3));     //true
		System.out.println(n==n3);            //false
		
 
	}
	
	public static void main(String[] args) {
		Rou de = new Rou();
		de.verify();
	}
}
